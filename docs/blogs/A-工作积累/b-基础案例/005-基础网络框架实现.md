```java
import java.util.ArrayList;
import java.util.List;

import okhttp3.Interceptor;
import okhttp3.logging.HttpLoggingInterceptor;
import retrofit2.converter.gson.GsonConverterFactory;

public class Net {

    private static NetOrdinary netOrdinary; // 通用的网络请求 普通 ordinary

    /**
     * 接口文件的字节码类
     */
    public static <T> T create(Class<T> service) {
        if (netOrdinary == null) {
            netOrdinary = NetOrdinary.create(networkConfig -> null);
        }
        return netOrdinary.create(service);
    }

    /**
     * 初始化的时候，设置了拦截器
     */
    public static void initNet(String url) {
        netOrdinary = NetOrdinary.create(builder -> {
            List<Interceptor> list = new ArrayList<>();

            HttpAllHeadInterceptor allHeadInterceptor = new HttpAllHeadInterceptor();
            HttpLoggingInterceptor loggingInterceptor = new HttpLoggingInterceptor().setLevel(HttpLoggingInterceptor.Level.NONE);

            list.add(allHeadInterceptor);
            list.add(loggingInterceptor);

            builder.setHostUrl(url).setInterceptors(list).setConverter(GsonConverterFactory.create());
            return null;
        });
    }

    /**
     * 是否初始化过
     */
    public static boolean isNetInit() {
        return netOrdinary != null;
    }
}
```

```java

import java.util.concurrent.TimeUnit;

import kotlin.Unit;
import kotlin.jvm.functions.Function1;
import okhttp3.Interceptor;
import okhttp3.OkHttpClient;
import retrofit2.Retrofit;
import retrofit2.converter.gson.GsonConverterFactory;

// 访问类
public class NetOrdinary {

    private final Retrofit mRetrofit;

    public static NetOrdinary create(Function1<NetworkConfig, Unit> action) {
        NetworkConfig config = NetworkConfig.build(action);
        return new NetOrdinary(config);
    }

    private NetOrdinary(NetworkConfig config) {
        OkHttpClient.Builder clientBuilder = new OkHttpClient.Builder()
                .connectTimeout(30, TimeUnit.SECONDS)
                .readTimeout(30, TimeUnit.SECONDS)
                .writeTimeout(30, TimeUnit.SECONDS);

        if (config.getInterceptors() != null) {
            for (Interceptor interceptor : config.getInterceptors()) {
                clientBuilder.addInterceptor(interceptor);
            }
        }

        OkHttpClient client = clientBuilder.build();

        String hostUrl = config.getHostUrl();
        if (TextUtils.isEmpty(hostUrl)) {
            throw new HostURLException("必须在NetworkConfig传入HostURL.此时此刻HostURL为空");
        }

        mRetrofit = new Retrofit.Builder()
                .client(client)
                .baseUrl(hostUrl)
                .addConverterFactory(config.getConverter() != null ? config.getConverter() : GsonConverterFactory.create())
                .addCallAdapterFactory(new NetworkFactory())
                .build();
    }

    public <T> T create(Class<T> service) {
        return mRetrofit.create(service);
    }
}
```

位置 `operator.base.NetworkFactory.java`

```java
import java.lang.annotation.Annotation;
import java.lang.reflect.ParameterizedType;
import java.lang.reflect.Type;
import java.util.concurrent.Executor;

import retrofit2.Call;
import retrofit2.CallAdapter;
import retrofit2.Retrofit;

final class NetworkFactory extends CallAdapter.Factory {
    private final Executor callbackExecutor;

    NetworkFactory() {
        callbackExecutor = new MainThreadExecutor();
    }

    @Override
    public CallAdapter<?, ?> get(Type returnType, Annotation[] annotations, Retrofit retrofit) {
        if (getRawType(returnType) != NetworkCall.class) {
            return null;
        }
        if (!(returnType instanceof ParameterizedType)) {
            throw new IllegalArgumentException(
                    "Call return type must be parameterized as Call<Foo> or Call<? extends Foo>");
        }
        final Type responseType = getParameterUpperBound(0, (ParameterizedType) returnType);

        final Executor executor = callbackExecutor;

        return new CallAdapter<Object, Call<?>>() {
            @Override
            public Type responseType() {
                return responseType;
            }

            @Override
            public Call<Object> adapt(Call<Object> call) {
                return new NetworkCallImpl<>(executor, call);
            }
        };
    }

    static final class MainThreadExecutor implements Executor {
        private final Handler handler = new Handler(Looper.getMainLooper());

        @Override
        public void execute(Runnable r) {
            handler.post(r);
        }
    }
}
```

```java
        return message;
```

位置 `operator.impl.EnqueueCallImpl.java`

```java
import androidx.annotation.NonNull;
import androidx.lifecycle.Lifecycle;
import androidx.lifecycle.LifecycleEventObserver;
import androidx.lifecycle.LifecycleOwner;
import androidx.lifecycle.MutableLiveData;
import androidx.lifecycle.Observer;

import java.net.UnknownHostException;
import java.util.concurrent.Executor;

import retrofit2.Call;
import retrofit2.Callback;
import retrofit2.Response;


/**
 * 真正执行请求的实现类
 */
class EnqueueCallImpl<T> implements EnqueueCall<T> {
    final private Executor callbackExecutor;
    final Call<T> delegate;

    MutableLiveData<T> onSuccess;
    MutableLiveData<NetWorkException> onFailure;
    MutableLiveData<NetworkState> onState;

    EnqueueCallImpl(Executor executor, Call<T> delegate) {
        callbackExecutor = executor;
        this.delegate = delegate;
    }

    @Override
    public EnqueueCall<T> success(@NonNull Observer<T> success) {
        if (onSuccess == null) {
            onSuccess = new MutableLiveData<>();
        }
        onSuccess.observeForever(success);
        return this;
    }

    @Override
    public EnqueueCall<T> success(@NonNull LifecycleOwner owner, @NonNull Observer<T> success) {
        if (onSuccess == null) {
            onSuccess = new MutableLiveData<>();
        }
        onSuccess.observe(owner, success);
        return this;
    }

    @Override
    public EnqueueCall<T> failure(@NonNull Observer<NetWorkException> failure) {
        if (onFailure == null) {
            onFailure = new MutableLiveData<>();
        }
        onFailure.observeForever(failure);
        return this;
    }

    @Override
    public EnqueueCall<T> failure(@NonNull LifecycleOwner owner, @NonNull Observer<NetWorkException> failure) {
        if (onFailure == null) {
            onFailure = new MutableLiveData<>();
        }
        onFailure.observe(owner, failure);
        return this;
    }

    @Override
    public EnqueueCall<T> owner(@NonNull LifecycleOwner owner) {
        owner.getLifecycle().addObserver(new LifecycleEventObserver() {
            @Override
            public void onStateChanged(@NonNull LifecycleOwner source, @NonNull Lifecycle.Event event) {
                if (event == Lifecycle.Event.ON_DESTROY) {
                    owner.getLifecycle().removeObserver(this);
                    cancel();
                }
            }
        });
        return this;
    }

    @Override
    public void enqueue() {

        if (onState != null) {
            onState.setValue(NetworkState.LOADING);
        }
        delegate.enqueue(new Callback<T>() {
            @Override
            public void onResponse(Call<T> call, Response<T> response) {
                T body = response.body();
                if (response.isSuccessful() && body != null) {
                    callbackExecutor.execute(() -> {
                        if (onState != null) {
                            onState.setValue(NetworkState.SUCCESS);
                        }
                        if (onSuccess != null) {
                            onSuccess.setValue(body);
                        }
                    });
                } else {
                    NetWorkException error = new NetWorkException("服务器错误");
                    error.setCode(response.code());
                    onFailure(call, error);
                }
            }

            @Override
            public void onFailure(Call<T> call, Throwable t) {
                callbackExecutor.execute(() -> {
                    if (onState != null) {
                        onState.setValue(NetworkState.ERROR);
                    }
                    if (onFailure != null) {
                        if (t instanceof NetWorkException) {
                            onFailure.setValue((NetWorkException) t);
                        } else if (t instanceof UnknownHostException) {
                            onFailure.setValue(new NetWorkException("网络错误"));
                        } else {
                            onFailure.setValue(new NetWorkException(t));
                        }
                    }
                });
            }
        });
    }

    void cancel() {
        delegate.cancel();
        onSuccess = null;
        onFailure = null;
        onState = null;
    }
}
```

```java
import androidx.annotation.NonNull;
import androidx.lifecycle.LifecycleOwner;
import androidx.lifecycle.Observer;

import java.io.IOException;
import java.util.Objects;
import java.util.concurrent.Executor;

import okhttp3.Request;
import okio.Timeout;
import retrofit2.Call;
import retrofit2.Callback;
import retrofit2.Response;

public final class NetworkCallImpl<T> implements NetworkCall<T> {
    final Executor callbackExecutor;
    final Call<T> delegate;
    EnqueueCallImpl<T> impl;

    public NetworkCallImpl(Executor executor, Call<T> delegate) {
        this.callbackExecutor = executor;
        this.delegate = delegate;
        impl = new EnqueueCallImpl<>(executor, delegate);
    }

    @Override
    public boolean isExecuted() {
        return delegate.isExecuted();
    }

    @Override
    public Response<T> execute() throws IOException {
        return delegate.execute();
    }

    /**
     * 不建议使用, 会导致下面方法无效
     * {@link EnqueueCall#success(Observer)}
     * {@link EnqueueCall#success(LifecycleOwner, Observer)}
     * {@link EnqueueCall#failure(Observer)}
     * {@link EnqueueCall#failure(LifecycleOwner, Observer)}
     * {@link EnqueueCall#success(LifecycleOwner, Observer)}
     * {@link EnqueueCall#owner(LifecycleOwner)}
     */
    @Override
    @Deprecated
    public void enqueue(Callback<T> callback) {
        Objects.requireNonNull(callback, "callback == null");

        delegate.enqueue(new Callback<T>() {
            @Override
            public void onResponse(Call<T> call, final Response<T> response) {
                callbackExecutor.execute(() -> {
                    if (delegate.isCanceled()) {
                        // Emulate OkHttp's behavior of throwing/delivering an IOException on
                        // cancellation.
                        callback.onFailure(NetworkCallImpl.this, new IOException("Canceled"));
                    } else {
                        callback.onResponse(NetworkCallImpl.this, response);
                    }
                });
            }

            @Override
            public void onFailure(Call<T> call, final Throwable t) {
                callbackExecutor.execute(() -> callback.onFailure(NetworkCallImpl.this, t));
            }
        });
    }

    @Override
    public void cancel() {
        impl.cancel();
        delegate.cancel();
    }

    @Override
    public boolean isCanceled() {
        return delegate.isCanceled();
    }

    @SuppressWarnings("CloneDoesntCallSuperClone") // Performing deep clone.
    @Override
    public Call<T> clone() {
        return new NetworkCallImpl<>(callbackExecutor, delegate.clone());
    }

    @Override
    public Request request() {
        return delegate.request();
    }

    @Override
    public Timeout timeout() {
        return delegate.timeout();
    }

    /***** 以下是拓展的方法 *****/
    @Override
    public NetworkCall<T> success(@NonNull Observer<T> success) {
        impl.success(success);
        return this;
    }

    @Override
    public NetworkCall<T> success(@NonNull LifecycleOwner owner, @NonNull Observer<T> success) {
        impl.success(owner, success);
        return this;
    }

    @Override
    public NetworkCall<T> failure(@NonNull Observer<NetWorkException> failure) {
        impl.failure(failure);
        return this;
    }

    @Override
    public NetworkCall<T> failure(@NonNull LifecycleOwner owner, @NonNull Observer<NetWorkException> failure) {
        impl.failure(owner, failure);
        return this;
    }


    @Override
    public NetworkCall<T> owner(@NonNull LifecycleOwner owner) {
        impl.owner(owner);
        return this;
    }

    @Override
    public void enqueue() {
        impl.enqueue();
    }

}
```
