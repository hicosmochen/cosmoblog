### 第03节 工厂类 FragmentFactory

```java
public class FragmentFactory {

    public static Fragment createFragment(Class<? extends Fragment> clazz) {
        Fragment result = null;

        switch (clazz.getName()) {
            case "com.cosmo.good.study.ui.fragment.FragmentA":
                result = new FragmentA();
                break;
            case "com.cosmo.good.study.ui.fragment.FragmentB":
                result = new FragmentB();
                break;
            case "com.cosmo.good.study.ui.fragment.FragmentC":
                result = new FragmentC();
                break;
            case "com.cosmo.good.study.ui.fragment.FragmentD":
                result = new FragmentD();
                break;
        }
        return result;
    }
}
```



### 第04节 适配器类

```java
import androidx.annotation.NonNull;
import androidx.fragment.app.Fragment;
import androidx.fragment.app.FragmentActivity;
import androidx.viewpager2.adapter.FragmentStateAdapter;

import java.util.List;

// 针对于viewPager2 的 Fragment 适配器类
public class DemoAdapter extends FragmentStateAdapter {

    private List<Fragment> fragments;

    public DemoAdapter(@NonNull FragmentActivity fragmentActivity, List<Fragment> fragments) {
        super(fragmentActivity);
        this.fragments = fragments;
    }

    @NonNull
    @Override
    public Fragment createFragment(int position) {
        return fragments.get(position);
    }

    @Override
    public int getItemCount() {
        return fragments == null ? 0 : fragments.size();
    }
}
```



### 第05节 界面  DemoActivity

```java
public class DemoActivity extends AppCompatActivity {

    private TabLayout mTabLayout;
    private ViewPager2 mViewPager;
    private TabLayoutMediator mMediator;

    private final String[] titleArray = new String[]{
            "选项卡1", "选项卡2", "选项卡3", "选项卡4"
    };


    @Override
    protected void onCreate(@Nullable Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_demo);

        mTabLayout = findViewById(R.id.tl_content);
        mViewPager = findViewById(R.id.vp_content);

        List<Fragment> fragments = new ArrayList<>();
        fragments.add(FragmentFactory.createFragment(FragmentA.class));
        fragments.add(FragmentFactory.createFragment(FragmentB.class));
        fragments.add(FragmentFactory.createFragment(FragmentC.class));
        fragments.add(FragmentFactory.createFragment(FragmentD.class));
        mViewPager.setAdapter(new DemoAdapter(this, fragments));
        mViewPager.setOffscreenPageLimit(3);


        mMediator = new TabLayoutMediator(mTabLayout, mViewPager, (tab, position) -> {
            tab.setText(titleArray[position]);
        });
        mMediator.attach();

        // 注册监听器
        mViewPager.registerOnPageChangeCallback(new ViewPager2.OnPageChangeCallback() {
            @Override
            public void onPageScrolled(int position, float positionOffset, int positionOffsetPixels) {
                super.onPageScrolled(position, positionOffset, positionOffsetPixels);
            }

            @Override
            public void onPageSelected(int position) {
                super.onPageSelected(position);
                mTabLayout.selectTab(mTabLayout.getTabAt(position));
            }

            @Override
            public void onPageScrollStateChanged(int state) {
                super.onPageScrollStateChanged(state);
            }
        });


        // TabLayout 需要添加监听器
        mTabLayout.addOnTabSelectedListener(new TabLayout.OnTabSelectedListener() {
            @Override
            public void onTabSelected(TabLayout.Tab tab) {
                for (int i = 0; i < titleArray.length; i++) {
                    if (TextUtils.equals(titleArray[i], tab.getText())) {
                        mViewPager.setCurrentItem(i);
                    }
                }
            }

            @Override
            public void onTabUnselected(TabLayout.Tab tab) {

            }

            @Override
            public void onTabReselected(TabLayout.Tab tab) {

            }
        });
    }
}
```
