### 第04节 mvp.presenter 包

包名称:  `com.cosmo.good.study.mvp.presenter`

```java
import android.os.Handler;
import android.os.Looper;
import android.util.Log;

import com.cosmo.good.study.mvp.base.BaseMvpPresenter;
import com.cosmo.good.study.mvp.bean.User;
import com.cosmo.good.study.mvp.model.UserLoginModel;
import com.cosmo.good.study.mvp.view.UserLoginView;

public class UserLoginPresenter extends BaseMvpPresenter<UserLoginView> {

    private UserLoginModel iUserModel;

    public UserLoginPresenter(UserLoginView baseView) {
        super(baseView);
    }


    /**
     * 登录操作
     */
    public void login() {
        //进度框
        mView.showLoading();

        // 模拟延时的操作
        new Handler(Looper.getMainLooper()).postDelayed(() -> {

            iUserModel.login(mView.getUserName(), mView.getPassword(),
                    success -> {
                        mView.toMainActivity(success);
                        mView.hideLoading();
                        Log.i("UserLoginPresenter", "隐藏进度条");
                    },
                    fail -> {
                        mView.showFailedError();
                        mView.hideLoading();
                        Log.i("UserLoginPresenter", "隐藏进度条");
                    });

        }, 2000L);

    }

    /**
     * 清空操作
     */
    public void clear() {
        mView.clearUserName();
        mView.clearPassword();
    }

    @Override
    protected void initModel() {
        iUserModel = (username, password, success, fail) -> {
            if ("admin".equals(username) && "123".equals(password)) {
                //执行成功的业务逻辑
                User u = new User();
                u.setUsername(username);
                u.setPassword(password);
                success.callBack(u);
            } else {
                //执行失败的业务逻辑
                fail.callBack("");
            }
        };
    }
}
```

## 第三章 activity相关

### 第01节 ui.activity 包

包名称:  `com.cosmo.good.study.ui.activity `

```java
import android.os.Bundle;
import android.util.Log;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.ProgressBar;
import android.widget.TextView;

import androidx.appcompat.app.AppCompatActivity;

import com.cosmo.good.study.R;
import com.cosmo.good.study.mvp.bean.User;
import com.cosmo.good.study.mvp.presenter.UserLoginPresenter;
import com.cosmo.good.study.mvp.view.UserLoginView;

public class MainActivity extends AppCompatActivity implements UserLoginView {

    private Button button1;
    private Button button2;
    private EditText editText1;
    private EditText editText2;
    private TextView textView;
    private ProgressBar progressbar;

    private UserLoginPresenter presenter = new UserLoginPresenter(this);


    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        button1 = findViewById(R.id.button1);
        button2 = findViewById(R.id.button2);
        editText1 = findViewById(R.id.editTextText1);
        editText2 = findViewById(R.id.editTextText2);
        progressbar = findViewById(R.id.progressbar);
        textView = findViewById(R.id.textview);

        button1.setOnClickListener(v -> {
            presenter.login();
        });

        button2.setOnClickListener(v -> {
            presenter.clear();
        });
    }

    @Override
    public String getUserName() {
        String username = editText1.getText().toString();
        showMessage("获取账号信息 getUserName: " + username);
        return username;
    }

    @Override
    public String getPassword() {
        String password = editText2.getText().toString();
        showMessage("获取密码信息 getPassword: " + password);
        return password;
    }

    @Override
    public void showLoading() {

        textView.setVisibility(View.GONE);
        progressbar.setVisibility(View.VISIBLE);
    }

    @Override
    public void hideLoading() {
         progressbar.setVisibility(View.GONE);
         textView.setVisibility(View.VISIBLE);
    }

    @Override
    public void clearUserName() {
        editText1.setText("");
        showMessage("清除账号 clearUserName");
    }

    @Override
    public void clearPassword() {
        editText2.setText("");
        showMessage("清除密码 clearPassword");
    }

    @Override
    public void toMainActivity(User user) {
        showMessage(user.toString());
    }

    @Override
    public void showFailedError() {
        showMessage("显示错误 showFailedError");
    }

    private void showMessage(String msg) {
        String message = textView.getText().toString();
        message += "\n";
        message += msg;
        textView.setText(message);
    }

}
```
