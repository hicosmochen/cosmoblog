## 第一章 前言

### 第01节 提出问题

开发过程中遇到的问题

```
问题： 协同开发的过程中, 经常会遇到一些需要前后端同时开发的情况。这种情况下, 大多数接口是没有连通的, 但是会给出 接口模型。

这种情况下, 我们不可能等待后端开发完毕之后, 前端才去开发, 否则前端开发会滞后, 后续带来了无限量的加班情况。
那么这种情况下, 我们应该怎么办呢? 
```

> 接口模型 
>
> 1、访问地址URL
>
> 2、接口的请求方式
>
> 3、接口请求需要什么参数
>
> 4、接口会响应到什么结果







### 第02节 解决问题

如何解决协同开发过程中的问题

```
第一步: 前端开发人员 和 后端开发人员 需要定义 统一的规范接口 (接口模型)
第二步: 前端开发人员 需要根据接口模型, 进行前期的访问数据模拟, 模拟网络请求访问, 得到需要的结果
```

如何进行前期的数据模拟操作

```
我们可以采用网络框架中的拦截器, 提前拦截后续的请求和响应。

1、拦截请求
2、通过模拟的接口地址, 访问模拟的响应数据
3、将模拟响应数据, 返回给请求方法
4、对模拟的响应结果, 进行UI的交互

备注: 后期只需要修改 网络框架模拟的拦截器, 让模拟的拦截器不生效, 执行真实的网络访问即可。
```

```java
import java.util.HashMap;
import java.util.Map;

// 映射配置表
public class ConfigureTable {

    // 键: 接口当中的 URL 地址
    // 值: 本地mock的响应 json 文件的地址
    private final Map<String, String> map = new HashMap<>();

    private static volatile ConfigureTable instance;

    private ConfigureTable() {
        init();
    }

    public static ConfigureTable getInstance() {
        if (instance == null) {
            synchronized (ConfigureTable.class) {
                if (instance == null) {
                    instance = new ConfigureTable();
                }
            }
        }
        return instance;
    }

    // 配置样式
    private void init() {
        map.put("/api/products", "mock_responses/products.json");
        map.put("/api/user", "mock_responses/user_profile.json");
    }

    public String getMockResponsePath(String urlKey) {
        return map.getOrDefault(urlKey, "");
    }
}
```

```java
import android.content.Context;

import okhttp3.OkHttpClient;
import retrofit2.Retrofit;
import retrofit2.adapter.rxjava3.RxJava3CallAdapterFactory;
import retrofit2.converter.gson.GsonConverterFactory;

// 网络请求类
public class ApiClient {

    // 是否是 Mock 环境, 如果是 Mock 环境下, 则拦截器生效, 否则不生效
    private static final boolean MOCK_ENVIRONMENT = true;

    private static Retrofit retrofit = null;

    public static Retrofit getClient(Context context) {
        if (retrofit == null) {
            OkHttpClient okHttpClient = new OkHttpClient.Builder()
                    .addInterceptor(new MockInterceptor(context, MOCK_ENVIRONMENT))    // 在这里添加的拦截器, 进行操作处理
                    .build();

            retrofit = new Retrofit.Builder()
                    .baseUrl(ApiService.BASE_URL)
                    .client(okHttpClient)
                    .addConverterFactory(GsonConverterFactory.create())
                    .addCallAdapterFactory(RxJava3CallAdapterFactory.create())
                    .build();
        }
        return retrofit;
    }
}
```

模拟拦截器

```java
import android.content.Context;
import android.text.TextUtils;

import androidx.annotation.NonNull;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.nio.charset.StandardCharsets;

import okhttp3.Interceptor;
import okhttp3.MediaType;
import okhttp3.Protocol;
import okhttp3.Response;
import okhttp3.ResponseBody;

// 模拟拦截器
public class MockInterceptor implements Interceptor {

    private final Context context;
    private final boolean isMockEnvironment;
    private static final int CODE_SUCCESS = 200;
    private static final int CODE_FAIL = 404;
    private static final String HEAD_APPLICATION_JSON = "application/json";
    private static final String HEAD_CONTENT_TYPE = "content-type";
    private static final String MESSAGE_ERROR_RESPONSE = "{\"error\":\"No mock data found for this endpoint\"}";

    public MockInterceptor(Context context, boolean isMockEnvironment) {
        this.context = context;
        this.isMockEnvironment = isMockEnvironment;
    }

    @NonNull
    @Override
    public Response intercept(@NonNull Chain chain) throws IOException {
        // 如果不是  mock 环境下, 不做拦截, 直接返回
        if (!isMockEnvironment) {
            return chain.proceed(chain.request());
        }

        int code;
        String responseString;
        String url = chain.request().url().encodedPath();
        String path = ConfigureTable.getInstance().getMockResponsePath(url);

        // 根据请求路径返回不同的模拟数据, 如果路径为空的情况下, 则返回错误 404
        if (TextUtils.isEmpty(path)) {
            code = CODE_FAIL;
            responseString = MESSAGE_ERROR_RESPONSE;
        } else {
            code = CODE_SUCCESS;
            responseString = readAssets(context, path);
        }

        return new Response.Builder()
                .code(code)
                .message("OK")
                .request(chain.request())
                .protocol(Protocol.HTTP_1_1)
                .body(ResponseBody.create(MediaType.parse(HEAD_APPLICATION_JSON), responseString))
                .addHeader(HEAD_CONTENT_TYPE, HEAD_APPLICATION_JSON)
                .build();
    }


    public static String readAssets(Context context, String fileName) {
        String result = "";
        try {
            InputStream is = context.getAssets().open(fileName);
            BufferedReader br = new BufferedReader(new InputStreamReader(is, StandardCharsets.UTF_8));
            StringBuilder sb = new StringBuilder();
            String line;
            while ((line = br.readLine()) != null) {
                sb.append(line);
            }
            br.close();
            result = sb.toString();
        } catch (IOException e) {
            return result;
        }
        return result;
    }
}
```

```java
import android.content.Context;
import android.os.Bundle;
import android.widget.Toast;

import androidx.appcompat.app.AppCompatActivity;
import androidx.appcompat.widget.AppCompatTextView;

import io.reactivex.rxjava3.android.schedulers.AndroidSchedulers;
import io.reactivex.rxjava3.annotations.NonNull;
import io.reactivex.rxjava3.core.Observer;
import io.reactivex.rxjava3.disposables.Disposable;
import io.reactivex.rxjava3.schedulers.Schedulers;


public class MainActivity extends AppCompatActivity {

    private Context context = MainActivity.this;
    private static final String TAG = MainActivity.class.getSimpleName();

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        AppCompatTextView textView = findViewById(R.id.text_view);

        textView.setOnClickListener(view -> visitWeb());
    }

    // 访问网络: 采用 RxJava 
    private void visitWeb() {
        ApiClient.getClient(this).create(ApiService.class).getProduct()
                .subscribeOn(Schedulers.io())
                .observeOn(AndroidSchedulers.mainThread())
                .subscribe(new Observer<>() {
                    @Override
                    public void onSubscribe(@NonNull Disposable d) {

                    }

                    @Override
                    public void onNext(@NonNull ProductBean productBean) {
                        Toast.makeText(context, "bean: " + productBean, Toast.LENGTH_LONG).show();
                    }

                    @Override
                    public void onError(@NonNull Throwable error) {
                        Toast.makeText(context, "error: " + error, Toast.LENGTH_LONG).show();
                    }

                    @Override
                    public void onComplete() {

                    }
                });
    }
}
```
