## 第一章 前言

### 第01节 提出问题

问题描述

```
问题1: 
	在 Android 开发过程中, 难免会遇到出现异常的情况, 如果出现异常之后, 直接闪退, 用户体验非常不好。
	
问题2:
	如果系统出现了崩溃的情况, 作为开发者, 需要快速定位到出现问题的原因, 这种情况下, 需要立即拿到崩溃的日志
	便于快速定位到问题, 快速解决问题。
```





### 第02节 分析问题

分析问题

```
在 Android 代码当中, 如果出现了问题, 会执行一段代码的逻辑。
将会执行下面一段代码逻辑:
	1、接口 Thread.UncaughtExceptionHandler 
	2、方法 void uncaughtException(@RecentlyNonNull Thread var1, @RecentlyNonNull Throwable var2);
	
那么我们可以重写这段代码逻辑, 针对于出现的问题, 进行抓捕, 缓存, 甚至发送到服务器当中。
```

位置：  `src/main/java/com.cosmo.good.study.util.CrashHandler.java`

```java
import android.app.Activity;
import android.content.Context;
import android.content.Intent;
import android.content.pm.PackageInfo;
import android.content.pm.PackageManager;
import android.os.Build;
import android.os.Environment;
import android.os.Looper;
import android.text.TextUtils;
import android.util.Log;

import java.io.File;
import java.io.FileOutputStream;
import java.io.PrintWriter;
import java.io.StringWriter;
import java.io.Writer;
import java.lang.reflect.Field;
import java.text.DateFormat;
import java.text.SimpleDateFormat;
import java.util.Date;
import java.util.HashMap;
import java.util.LinkedHashMap;
import java.util.Locale;
import java.util.Map;
import java.util.Set;
import java.util.TreeMap;

/****
 *
 * @author cosmo
 *
 * @since 2023/9/7 10:58
 *
 * @apiNote UncaughtException处理类, 当程序发生Uncaught异常的时候, 由该类来接管程序, 并记录发送错误报告.
 */
public class CrashHandler implements Thread.UncaughtExceptionHandler {

    public static final String TAG = "CrashHandler";

    // 系统默认的UncaughtException处理类
    private Thread.UncaughtExceptionHandler mDefaultHandler;
    // CrashHandler实例
    private static CrashHandler INSTANCE = new CrashHandler();
    // 程序的Context对象
    private Context mContext;
    private long fileLiveTime;
    private long fileMaxSize;
    private final long DEFAULT_FILE_LIVE_TIME = 1000 * 60 * 60 * 24 * 3;  // 默认的最大存活时间设置为3天
    private final long DEFAULT_FILE_MAX_SIZE = 1024 * 1024 * 32;  // 默认的最大缓存日志的大小为 32MB

    private Class<? extends Activity> clazzErrorActivity;

    public interface Key {
        String FILE_LIVE_TIME = "FILE_LIVE_TIME";
        String FILE_MAX_SIZE = "FILE_MAX_SIZE";
        String ERROR_ACTIVITY_PATH = "ERROR_ACTIVITY_PATH";
    }

    // 定义一些预设的参数
    private HashMap<String, String> paramMap;


    //用来存储设备信息和异常信息
    private Map<String, String> infos = new LinkedHashMap<>();

    //用于格式化日期,作为日志文件名的一部分
    private DateFormat formatter = new SimpleDateFormat("yyyy-MM-dd-HH-mm-ss");

    /**
     * 保证只有一个CrashHandler实例
     */
    private CrashHandler() {
    }

    /**
     * 获取CrashHandler实例 ,单例模式
     */
    public static CrashHandler getInstance() {
        return INSTANCE;
    }

    /**
     * 初始化
     *
     * @param context
     */
    public CrashHandler init(Context context) {
        mContext = context;
        //获取系统默认的UncaughtException处理器
        mDefaultHandler = Thread.getDefaultUncaughtExceptionHandler();
        //设置该CrashHandler为程序的默认处理器
        Thread.setDefaultUncaughtExceptionHandler(this);
        return this;
    }

    public void initParamMap(HashMap<String, String> paramMap) {
        this.paramMap = paramMap;
        String fileLiveTime = paramMap.get(Key.FILE_LIVE_TIME);
        String fileMaxSize = paramMap.get(Key.FILE_MAX_SIZE);
        String errorActivityPath = paramMap.get(Key.ERROR_ACTIVITY_PATH);
        // 在这里进行数据的处理
        try {
            this.fileLiveTime = Long.parseLong(fileLiveTime);
        } catch (Exception ignored) {
            this.fileLiveTime = DEFAULT_FILE_LIVE_TIME;
        }
        try {
            this.fileMaxSize = Long.parseLong(fileMaxSize);
        } catch (Exception ignored) {
            this.fileMaxSize = DEFAULT_FILE_MAX_SIZE;
        }
        if (TextUtils.isEmpty(fileLiveTime)) {
            this.fileLiveTime = DEFAULT_FILE_LIVE_TIME;
        }
        if (TextUtils.isEmpty(fileMaxSize)) {
            this.fileMaxSize = DEFAULT_FILE_MAX_SIZE;
        }
        try {
            clazzErrorActivity = (Class<? extends Activity>) Class.forName(errorActivityPath);
        } catch (ClassNotFoundException e) {

        }
    }

    /**
     * 当UncaughtException发生时会转入该函数来处理
     */
    @Override
    public void uncaughtException(Thread thread, Throwable ex) {
        if (!handleException(ex) && mDefaultHandler != null) {
            //如果用户没有处理则让系统默认的异常处理器来处理
            mDefaultHandler.uncaughtException(thread, ex);
        } else {
            try {
                Thread.sleep(1000);
            } catch (InterruptedException e) {
                Log.e(TAG, "error : ", e);
            }
            //退出程序
            android.os.Process.killProcess(android.os.Process.myPid());
            System.exit(1);
        }
    }

    private void jumpToErrorActivity() {
        Intent intent = new Intent();
        intent.setClass(mContext, clazzErrorActivity);
        intent.setFlags(Intent.FLAG_ACTIVITY_CLEAR_TASK | Intent.FLAG_ACTIVITY_NEW_TASK);
        mContext.startActivity(intent);
    }

    /**
     * 自定义错误处理,收集错误信息 发送错误报告等操作均在此完成.
     *
     * @param ex
     * @return true:如果处理了该异常信息;否则返回false.
     */
    private boolean handleException(Throwable ex) {
        if (ex == null) {
            return false;
        }
        //使用Toast来显示异常信息
        new Thread() {
            @Override
            public void run() {
                Looper.prepare();
                jumpToErrorActivity();
                Looper.loop();
            }
        }.start();
        //收集设备参数信息
        collectDeviceInfo(mContext);
        //保存日志文件
        String fileName = saveCrashInfo2File(ex);
        return true;
    }


    /**
     * 收集设备参数信息
     *
     * @param ctx
     */
    private void collectDeviceInfo(Context ctx) {
        try {
            PackageManager pm = ctx.getPackageManager();
            PackageInfo pi = pm.getPackageInfo(ctx.getPackageName(), PackageManager.GET_ACTIVITIES);
            if (pi != null) {
                String appName = pm.getApplicationLabel(pi.applicationInfo).toString();
                String versionName = pi.versionName == null ? "null" : pi.versionName;
                String versionCode = pi.versionCode + "";
                String packageName = pi.packageName;
                String firstInstallTime = formatTime(pi.firstInstallTime);
                String lastUpdateTime = formatTime(pi.lastUpdateTime);

                infos.put("appName", appName);
                infos.put("versionName", versionName);
                infos.put("versionCode", versionCode);
                infos.put("packageName", packageName);
                infos.put("firstInstallTime", firstInstallTime);
                infos.put("lastUpdateTime", lastUpdateTime);
            }
        } catch (Exception e) {
            Log.e(TAG, "an error occured when collect package info", e);
        }
        Field[] fields = Build.class.getDeclaredFields();
        for (Field field : fields) {
            try {
                field.setAccessible(true);
                infos.put(field.getName(), field.get(null).toString());
                Log.d(TAG, field.getName() + " : " + field.get(null));
            } catch (Exception e) {
                Log.e(TAG, "an error occured when collect crash info", e);
            }
        }
    }


    // 格式化时间
    private String formatTime(long time) {
        String format = "yyyy-MM-dd HH:mm:ss SSS";
        SimpleDateFormat sdf = new SimpleDateFormat(format, Locale.getDefault());
        return sdf.format(new Date(time));
    }

    /**
     * 保存错误信息到文件中
     *
     * @param ex
     * @return 返回文件名称, 便于将文件传送到服务器
     */
    private String saveCrashInfo2File(Throwable ex) {

        StringBuilder sb = new StringBuilder();
        for (Map.Entry<String, String> entry : infos.entrySet()) {
            String key = entry.getKey();
            String value = entry.getValue();
            sb.append(key).append(" = ").append(value).append("\n");
        }

        Writer writer = new StringWriter();
        PrintWriter printWriter = new PrintWriter(writer);
        ex.printStackTrace(printWriter);
        Throwable cause = ex.getCause();
        while (cause != null) {
            cause.printStackTrace(printWriter);
            cause = cause.getCause();
        }
        printWriter.close();
        String result = writer.toString();
        sb.append(result);
        try {
            long timestamp = System.currentTimeMillis();
            String time = formatter.format(new Date());
            // 创建文件夹和文件信息
            String fileName = "crash-" + time + "-" + timestamp + ".log";
            File dir = null;
            if (Environment.getExternalStorageState().equals(Environment.MEDIA_MOUNTED)) {
                String path = mContext.getObbDir().getPath() + "/crash/";
                dir = new File(path);
                if (!dir.exists()) {
                    dir.mkdirs();
                }
            }
            // 查看文件夹当中文件的数量, 如果数量超过阈值, 则删除文件夹当中的数据
            if (dir != null) {
                // 处理阈值数据的文件信息
                deleteOldErrorFile(dir);
                // 写数据到文件当中
                FileOutputStream fos = new FileOutputStream(new File(dir, fileName));
                fos.write(sb.toString().getBytes());
                fos.close();
            }
            return fileName;
        } catch (Exception e) {
            Log.e(TAG, "an error occured while writing file...", e);
        }
        return null;
    }

    // 下面的是采用 LRU 算法, 超过阈值之后, 根据最久远的时间删除掉缓存的数据, 删除之后, 再根据最后一次使用时间来删除最近的数据
    private void deleteOldErrorFile(File dir) {
        // 默认情况下的标记值 0
        int loopFlag = 0;
        while (true) {
            File[] files = dir.listFiles();
            // 获取文件夹下面, 所有文件的数量, 如果不存在, 那么直接返回结果, 不做后续的运算
            if (files == null || files.length == 0) {
                break;
            }
            // 计算总的大小
            long total = 0;
            for (File errorFile : files) {
                if (errorFile != null) {
                    total += errorFile.length();
                }
            }
            // 到达了最大阈值的情况下
            if (total >= fileMaxSize) {
                if (loopFlag == 0) {
                    // 如果 loopFlag==0 表示是第一轮的循环操作, 如果是第一轮的循环操作, 那么需要进行超过最大修改时间的删除操作
                    // 记录当前的情况为  loopFlag = 1;
                    loopFlag = 1;
                } else {
                    // 如果 loopFlag!=0 表示是第N轮的循环操作, 如果是第N轮的循环操作, 那么需要做每次删除最后一个数据的操作
                    // 记录当前的情况为 loopFlag = 2;
                    loopFlag = 2;
                }
            }
            // 如果当前记录的 loopFlag == 0 表示没有进入到上面超过最大阈值的情况, 那么直接跳出死循环了。
            if (loopFlag == 0) {
                break;
            }
            // 如果当前记录的 loopFlag != 0 表示当前已经进入到了最大阈值的情况, 需要针对于数据进行处理
            // 首选需要按照时间的顺序, 给最大操作阈值进行排序的操作。 （目前是 t1 - t2） 时间升序排列
            TreeMap<Long, FileEntry> treeMap = new TreeMap<>(Long::compare);
            for (int i = 0; i < files.length; i++) {
                FileEntry fileEntry = new FileEntry(files[i].getAbsolutePath(), files[i].length(), files[i].lastModified(), files[i]);
                treeMap.put(files[i].lastModified(), fileEntry);
            }
            // 第1轮, 删除超过存活时间的数据
            Set<Map.Entry<Long, FileEntry>> entries = treeMap.entrySet();

            // 如果当前的数据值为 2 那么表示是第二轮的操作, 这里就是需要删除掉第一位数字, 集合TreeMap当中的数据越靠前, 表示距离当前时间越久远。
            if (loopFlag == 2) {
                for (Map.Entry<Long, FileEntry> entry : entries) {
                    FileEntry value = entry.getValue();
                    boolean delete = value.file.delete();
                    loopFlag = 3;
                    break;
                }
            }
            // 如果执行到这里 loopFlag == 3 表示目前进行的是第二轮操作完毕之后的结果, 那么需要继续去判断操作。
            if (loopFlag == 3) {
                continue;
            }
            // 如果执行到这里 loopFlag != 3 那么表示当前的 loopFlag == 1 这里需要按照时间的超过阈值的需要删除掉
            for (Map.Entry<Long, FileEntry> entry : entries) {
                Long key = entry.getKey();
                FileEntry value = entry.getValue();
                if (System.currentTimeMillis() - key >= fileLiveTime) {
                    // 删除该文件
                    boolean delete = value.file.delete();
                    // 当文件进行了删除之后, 设置 loopFlag = 2 表示需要进行第二轮循环的操作
                    loopFlag = 2;
                }
            }
        }
    }

    private static class FileEntry {
        String filePath;
        long fileSize;
        long fileLastModified;
        File file;

        public FileEntry(String filePath, long fileSize, long fileLastModified, File file) {
            this.filePath = filePath;
            this.fileSize = fileSize;
            this.fileLastModified = fileLastModified;
            this.file = file;
        }
    }
}
```

```java
package com.cosmo.good.study.ui.activity;

import android.app.AlertDialog;
import android.content.Context;
import android.os.Bundle;

import androidx.annotation.Nullable;
import androidx.appcompat.app.AppCompatActivity;
import androidx.appcompat.widget.AppCompatTextView;

import com.cosmo.good.study.R;

import java.io.BufferedReader;
import java.io.File;
import java.io.FileInputStream;
import java.io.IOException;
import java.io.InputStreamReader;

/****
 *
 * @author cosmo
 *
 * @since 2023/9/7 17:26
 *
 * @apiNote 出错之后跳转的 Activity
 */
public class ErrorActivity extends AppCompatActivity {

    private AppCompatTextView errorTipsText;
    private Context context = ErrorActivity.this;

    @Override
    protected void onCreate(@Nullable Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_error);

        errorTipsText = findViewById(R.id.error_tips_text);
        errorTipsText.setOnClickListener(view -> {

            // 弹出对话框
            // 查找到最新的报错信息
            AlertDialog dialog = new AlertDialog.Builder(context).create();
            dialog.setMessage(readLogMessage());
            dialog.show();
        });
    }

    private String readLogMessage() {
        StringBuilder sb = new StringBuilder();
        // 获取到最新的报错信息
        File obbDir = context.getObbDir();
        File crashDir = new File(obbDir, "/crash/");
        File[] files = crashDir.listFiles();
        File lastFile = null;
        if (files != null) {
            lastFile = files[0];
            for (File childFile : files) {
                if (childFile.lastModified() > lastFile.lastModified()) {
                    lastFile = childFile;
                }
            }
        }
        // 判断内容
        if (lastFile != null) {
            BufferedReader br = null;
            try {
                // 读取数据
                br = new BufferedReader(new InputStreamReader(new FileInputStream(lastFile)));
                String line = null;
                while ((line = br.readLine()) != null) {
                    sb.append(line).append("\n");
                }
            } catch (IOException e) {
                throw new RuntimeException(e);
            } finally {
                if (br != null) {
                    try {
                        br.close();
                    } catch (IOException e) {
                        throw new RuntimeException(e);
                    }
                }
            }
        }
        return sb.toString();
    }
}
```

### 第04节 日志文件存储位置

我们日志文件的存储位置,  主要存储在 `obb` 文件夹当中。

```
1、 可以写入数据到 obb 文件夹当中
2、 可以读取数据从 obb 文件夹当中

当前 obb 文件夹所在的位置:
		storage/emulated/0/Android/obb/<packageName>/crash
		
对于上述文件目录的操作：
	A. 写入的操作, 见  CrashHandler.java  文件
	B. 读取的操作, 见  ErrorActivity.java 文件
	
另外: 对该文件的操作, 需要添加读写文件的权限处理
```
