> 需要注意的问题：
>
> 1、两个自定义视图分别是 `FastLetterLayoutView<T>`  和  `FastLetterIndexView` 
>
> 2、这里未做自定义属性的抽取，后期根据需要可以根据自定义属性抽取操作
>
> 3、整体的视图控制是  `FastLetterLayoutView` 和 对应的布局文件 `item_layout_normal.xml`

```java
import android.content.Context;
import android.content.res.Resources;
import android.graphics.Canvas;
import android.graphics.Color;
import android.graphics.Paint;
import android.graphics.Rect;
import android.text.TextUtils;
import android.util.AttributeSet;
import android.view.MotionEvent;
import android.view.View;

import androidx.annotation.Nullable;

public class FastLetterIndexView extends View {

    private static final String INDEX_NAME = "#ABCDEFGHIJKLMNOPQRSTUVWXYZ";
    private OnLetterUpdateListener listener;
    private Paint mPaint;
    private float cellHeight, viewWidth;
    private int touchIndex = -1;
    private int textColorDefault;
    private int textColorTouch;
    private float textSize = 14;


    public FastLetterIndexView(Context context) {
        this(context, null);
    }

    public FastLetterIndexView(Context context, @Nullable AttributeSet attrs) {
        this(context, attrs, 0);
    }

    public FastLetterIndexView(Context context, @Nullable AttributeSet attrs, int defStyleAttr) {
        super(context, attrs, defStyleAttr);
        mPaint = new Paint();

        mPaint.setFakeBoldText(true);

        mPaint.setTextSize(dp2px(textSize));
        mPaint.setAntiAlias(true);
        //获取文字被选中的颜色
        textColorDefault = Color.parseColor("#FF0000FF");
        textColorTouch = Color.parseColor("#FFFF0000");
    }

    @Override
    protected void onDraw(Canvas canvas) {
        for (int i = 0; i < INDEX_NAME.length(); i++) {
            String text = INDEX_NAME.substring(i, i + 1);
            //计算绘制字符的X方向起点
            int x = (int) (viewWidth / 2.0f - mPaint.measureText(text) / 2.0f);
            Rect bounds = new Rect();
            mPaint.getTextBounds(text, 0, text.length(), bounds);
            int textHeight = bounds.height();
            //计算绘制字符的Y方向起点
            int y = (int) (cellHeight / 2.0f + textHeight / 2.0f + i * cellHeight);
            mPaint.setColor(touchIndex == i ? textColorTouch : textColorDefault);
            canvas.drawText(text, x, y, mPaint);
        }
    }

    @Override
    public boolean onTouchEvent(MotionEvent event) {
        int index;
        switch (event.getAction()) {
            case MotionEvent.ACTION_DOWN:
                //计算当前触摸的字符索引
                index = (int) (event.getY() / cellHeight);
                if (index >= 0 && index < INDEX_NAME.length()) {
                    if (listener != null) {
                        listener.onLetterUpdate(INDEX_NAME.substring(index, index + 1));
                    }
                    touchIndex = index;
                }
                break;
            case MotionEvent.ACTION_MOVE:
                //计算当前触摸的字符索引
                index = (int) (event.getY() / cellHeight);
                if (index >= 0 && index < INDEX_NAME.length()) {
                    if (index != touchIndex) {
                        if (listener != null) {
                            listener.onLetterUpdate(INDEX_NAME.substring(index, index + 1));
                        }
                        touchIndex = index;
                    }
                }
                break;
        }
        invalidate();
        return true;
    }

    /**
     * 根据手机的分辨率从 dp 的单位 转成为 px(像素)
     */
    public static int dp2px(float dpValue) {
        float density  = Resources.getSystem().getDisplayMetrics().density;
        return (int) (0.5f + dpValue * Resources.getSystem().getDisplayMetrics().density);
    }


    // 外部修改数据
    public void setCurrentLatter(String latter){
        // 查询 latter 所在的数据集的位置
        for (int i = 0; i < INDEX_NAME.length(); i++) {
            if (TextUtils.equals(latter, String.valueOf(INDEX_NAME.charAt(i)).toUpperCase())){
                touchIndex = i;
                break;
            }
        }
        // 设置重绘
        invalidate();
    }

    @Override
    protected void onSizeChanged(int w, int h, int oldw, int oldh) {
        super.onSizeChanged(w, h, oldw, oldh);
        //得到当前控件的宽度
        viewWidth = getMeasuredWidth();
        int mHeight = getMeasuredHeight();
        //获取单个字符能够拥有的高度
        cellHeight = mHeight * 1.0F / INDEX_NAME.length();
    }

    public interface OnLetterUpdateListener {
        void onLetterUpdate(String letter);
    }

    public void setListener(OnLetterUpdateListener listener) {
        this.listener = listener;
    }

}
```

```java

import android.content.Context;
import android.graphics.Color;
import android.graphics.Typeface;
import android.os.Handler;
import android.os.Looper;
import android.text.TextUtils;
import android.util.AttributeSet;
import android.view.Gravity;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.LinearLayout;
import android.widget.PopupWindow;
import android.widget.TextView;
import android.widget.Toast;

import androidx.annotation.NonNull;
import androidx.annotation.Nullable;
import androidx.appcompat.widget.LinearLayoutCompat;
import androidx.recyclerview.widget.LinearLayoutManager;
import androidx.recyclerview.widget.RecyclerView;

import com.cosmo.good.study.R;
import com.github.stuxuhai.jpinyin.PinyinFormat;
import com.github.stuxuhai.jpinyin.PinyinHelper;
import com.google.gson.Gson;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

public class FastLetterLayoutView<T> extends LinearLayoutCompat {

    private RecyclerView recyclerView;
    private FastLetterIndexView fastIndexView;
    //主要用于展示数据的list
    private List<NameInfoModel<T>> currentList;
    //第一次加载之后缓存的数据
    private List<NameInfoModel<T>> cacheList;
    // 当前显示的文字
    private String currentCharset;
    //页面recyclerview的适配器
    private NameAdapter<T> adapter;
    //布局管理器
    private LinearLayoutManager layoutManager;
    private Context context;

    public FastLetterLayoutView(@NonNull Context context) {
        this(context, null);
    }

    public FastLetterLayoutView(@NonNull Context context, @Nullable AttributeSet attrs) {
        this(context, attrs, 0);
    }

    public FastLetterLayoutView(@NonNull Context context, @Nullable AttributeSet attrs, int defStyleAttr) {
        super(context, attrs, defStyleAttr);
        // 初始化视图
        initView(context, attrs, defStyleAttr);
        // 初始化数据
        initData();
        // 获取到第一行的数据
        findFirstLineData();
    }


    private void initView(Context context, AttributeSet attrs, int defStyleAttr) {
        this.context = context;
        View rootView = LayoutInflater.from(context).inflate(R.layout.view_fast_letter_layout, this);
        recyclerView = rootView.findViewById(R.id.recy_list);
        fastIndexView = rootView.findViewById(R.id.fastIndexView);
    }

    private void initData() {
        currentList = new ArrayList<>();
        cacheList = new ArrayList<>();
        List<NameInfoModel<T>> nameInfoModels = bindData(currentList);
        layoutManager = new LinearLayoutManager(context);
        recyclerView.setLayoutManager(layoutManager);
        adapter = new NameAdapter<T>(context, nameInfoModels);
        recyclerView.setAdapter(adapter);

        fastIndexView.setListener(letter -> {
            // 移动到置顶的位置
            moveToLetterPosition(letter);
            // 显示PopupWindow对话框
            showDialog(letter);
        });

        adapter.setOnViewListener(position -> {
            NameInfoModel<T> model = currentList.get(position);
            Toast.makeText(context, model.toString(), Toast.LENGTH_LONG).show();
        });
    }

    public void setDataSource(List<NameInfoModel<T>> dataSource) {
        currentList.clear();
        cacheList.clear();
        currentList.addAll(dataSource);
        bindData(currentList);
        adapter.notifyDataSetChanged();
    }


    // recyclerView 获取到第一行的数据, 如果展示出来的是第一行数据, 界面需要同步更新UI
    private void findFirstLineData() {
        // 获取到当前的位置
        LinearLayoutManager linearLayoutManager = (LinearLayoutManager) recyclerView.getLayoutManager();
        final int[] firstItemPositionOld = new int[]{layoutManager.findFirstCompletelyVisibleItemPosition()};

        recyclerView.addOnScrollListener(new RecyclerView.OnScrollListener() {
            @Override
            public void onScrolled(@NonNull RecyclerView recyclerView, int dx, int dy) {
                super.onScrolled(recyclerView, dx, dy);
                if (linearLayoutManager != null) {
                    int firstItemPositionNew = linearLayoutManager.findFirstCompletelyVisibleItemPosition();
                    if (firstItemPositionOld[0] != firstItemPositionNew) {
                        if (firstItemPositionNew == -1) {
                            return;
                        }
                        // 第一行的数据发生了变化, 需要处理一下
                        NameInfoModel nameInfoModel = currentList.get(firstItemPositionNew);
                        showDialog(nameInfoModel.getNameCharset());
                        // 重置数据
                        firstItemPositionOld[0] = firstItemPositionNew;
                    }
                }
            }
        });
    }

    // 弹框，显示的文本
    private void showDialog(String text) {
        if (TextUtils.equals(text, currentCharset)) {
            return;
        }
        // 重置左侧侧边栏的选项
        fastIndexView.setCurrentLatter(text);

        TextView textView = new TextView(context);
        textView.setLayoutParams(new LinearLayout.LayoutParams(ViewGroup.LayoutParams.MATCH_PARENT, ViewGroup.LayoutParams.MATCH_PARENT));
        textView.setGravity(Gravity.CENTER);
        textView.setTextSize(32);
        textView.setTypeface(Typeface.defaultFromStyle(Typeface.BOLD));
        textView.setTextColor(Color.WHITE);
        textView.setText(text);
        PopupWindow popupWindow = new PopupWindow(context);
        popupWindow.setWidth(300);
        popupWindow.setHeight(300);
        popupWindow.setContentView(textView);
        popupWindow.showAtLocation(recyclerView, Gravity.CENTER, 0, 0);
        new Handler(Looper.getMainLooper()).postDelayed(() -> {
            popupWindow.dismiss();
            currentCharset = text;
        }, 500L);
    }


    //滚动recyclerview
    private void moveToLetterPosition(String letter) {
        //这里主要是为了跳转到最顶端
        for (int i = 0; i < currentList.size(); i++) {
            NameInfoModel<T> cityInfoModel = currentList.get(i);
            if (cityInfoModel.getNameCharset().toUpperCase().equals(letter)) {
                layoutManager.scrollToPositionWithOffset(i, 0);
                return;
            }
        }
    }

    private List<NameInfoModel<T>> bindData(List<NameInfoModel<T>> dataList) {
        if (dataList != null) {
            for (NameInfoModel<T> infoModel : dataList) {
                cacheList.add(new NameInfoModel<T>(infoModel.getNameChinese(), infoModel.getData()));
            }
            //排序
            Collections.sort(cacheList, (o1, o2) -> o1.getPinyinAll().compareTo(o2.getPinyinAll()));
            this.currentList.clear();
            this.currentList.addAll(cacheList);
        }
        return currentList;
    }


    // ------------------------------------------------------------------------------------------
    // 需要存放的适配器的对象
    public static class NameAdapter<T> extends RecyclerView.Adapter<NameAdapter.NormalHolder> {

        private final List<NameInfoModel<T>> data;
        private final Context mContext;
        private NameAdapter.OnViewListener onViewListener;

        public NameAdapter(Context context, List<NameInfoModel<T>> data) {
            this.data = data;
            this.mContext = context;
        }

        public void setOnViewListener(NameAdapter.OnViewListener onViewListener) {
            this.onViewListener = onViewListener;
        }

        @NonNull
        @Override
        public NameAdapter.NormalHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
            View view = LayoutInflater.from(mContext).inflate(R.layout.item_layout_normal, parent, false);
            return new NameAdapter.NormalHolder(view);
        }

        @Override
        public void onBindViewHolder(NameAdapter.NormalHolder holder, final int position) {
            NameInfoModel<T> nameInfoModel = data.get(position);
            holder.tvCharset.setText(nameInfoModel.getNameCharset());
            holder.tvChinese.setText(nameInfoModel.getNameChinese());

            // 获取到前一项的首字符数据? 如果两行的首字符是一致的, 那么需要设置为隐藏, 否则展示
            holder.tvCharset.setVisibility(View.VISIBLE);

            if (position > 0) {
                NameInfoModel<T> nameInfoModelBefore = data.get(position - 1);
                if (TextUtils.equals(nameInfoModelBefore.getNameCharset(), nameInfoModel.getNameCharset())) {
                    holder.tvCharset.setVisibility(View.GONE);
                }
            }

            holder.tvChinese.setOnClickListener(view -> {
                if (onViewListener != null) {
                    onViewListener.onListener(position);
                }
            });
        }

        @Override
        public int getItemCount() {
            return data == null ? 0 : data.size();
        }


        static class NormalHolder extends RecyclerView.ViewHolder {
            TextView tvChinese;
            TextView tvCharset;

            public NormalHolder(View itemView) {
                super(itemView);
                tvCharset = itemView.findViewById(R.id.tv_charset);
                tvChinese = itemView.findViewById(R.id.tv_chinese);

            }
        }

        interface OnViewListener {
            void onListener(int position);
        }
    }


    // ------------------------------------------------------------------------------------------------------------
    public static class NameInfoModel<T>{

        private String nameChinese; // 用于显示完整的汉字
        private String nameCharset; // 用于排序的首字符
        private String pinyinAll;   // 用于排序的全拼音 这个是用于后面的排序以及搜索
        private T data;             // 附加的数据类型

        public static <T> NameInfoModel<T> create(String nameChinese, T data){
            return new NameInfoModel<>(nameChinese, data);
        }

        // 根据汉字得到对应的首字符和全拼
        private NameInfoModel(String nameChinese, T data) {
            this.nameChinese = nameChinese;
            try {
                this.pinyinAll = PinyinHelper.convertToPinyinString(nameChinese, "", PinyinFormat.WITHOUT_TONE);
                this.nameCharset = this.pinyinAll.substring(0, 1).toUpperCase();
                this.nameCharset = nameCharset.matches("[A-Z]") ? nameCharset : "#";
            } catch (Exception ignored) {

            }
            this.data = data;
        }

        public String getNameChinese() {
            return nameChinese;
        }

        public void setNameChinese(String nameChinese) {
            this.nameChinese = nameChinese;
        }

        public String getNameCharset() {
            return nameCharset;
        }

        public void setNameCharset(String nameCharset) {
            this.nameCharset = nameCharset;
        }

        public String getPinyinAll() {
            return pinyinAll;
        }

        public void setPinyinAll(String pinyinAll) {
            this.pinyinAll = pinyinAll;
        }

        public T getData() {
            return data;
        }

        public void setData(T data) {
            this.data = data;
        }

        @NonNull
        @Override
        public String toString() {
            return new Gson().toJson(this);
        }
    }
}
```

```java
import android.os.Bundle;

import androidx.appcompat.app.AppCompatActivity;

import com.cosmo.good.study.R;
import com.cosmo.good.study.common.view.FastLetterLayoutView;
import com.cosmo.good.study.data.Data;

import java.util.ArrayList;
import java.util.List;

public class DemoActivity extends AppCompatActivity {

    // 定义的泛型为附加的数据，根据附加的数据进行下一步的处理方案
    private FastLetterLayoutView<String> fastLetterLayoutView;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_demo);
        fastLetterLayoutView = findViewById(R.id.fast_letter_layout_view);

        String[] dataArray = Data.heroArray;

        // 设置城市效果
        List<FastLetterLayoutView.NameInfoModel<String>> list = new ArrayList<>();
        for (String str : dataArray) {
            list.add(FastLetterLayoutView.NameInfoModel.create(str, "hello"));
        }
        fastLetterLayoutView.setDataSource(list);
    }
}
```
