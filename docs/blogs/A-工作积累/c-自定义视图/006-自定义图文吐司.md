### 第01节 自定义的 Toast 类

```java
import android.content.Context;
import android.util.DisplayMetrics;
import android.view.Gravity;
import android.view.View;
import android.view.ViewGroup;
import android.view.WindowManager;
import android.widget.ImageView;
import android.widget.LinearLayout;
import android.widget.TextView;
import android.widget.Toast;

import com.blankj.utilcode.util.StringUtils;
import com.blankj.utilcode.util.Utils;
import com.cosmo.good.study.R;

/**
 *  自定义的 Toast 样式
 *
 *  <p>
 *      <b>01、基础用法</b>
 *      <div> 该文本显示在底部黄金分割点上 </div>
 *      <div> CommonToast.init().show("底部有点距离的吐司"); </div>
 *  </p>
 *  <p>
 *      <b>02、快速用法</b>
 *      <div> 快速展示在中间的水平图文信息 </div>
 *      <div> CommonToast.init().quickCenterHorizontal().show("中间的水平图文信息", R.mipmap.icon); </div>
 *      <div> 快速展示在中间的垂直图文信息 </div>
 *      <div> CommonToast.init().quickCenterVertical().show("中间的垂直图文信息", R.mipmap.icon); </div>
 *  </p>
 *  <p>
 *      <b>03、附加用法</b>
 *      <div> 附加用法可以设置偏移量 </div>
 *      <div> CommonToast.init().appendGravity(CommonToast.ToastGravity.top).appendOrientation(CommonToast.ToastOrientation.horizontal).appendOfferY(200).show("顶部的图文信息", R.mipmap.icon);</div>
 *  </p>
 *
 */
public class CommonToast {

    private static Context mContext;
    private volatile static CommonToast mInstance;
    private static Toast mToast;
    private View layout;
    private TextView viewText;
    private ImageView viewImage;

    private final LinearLayout.LayoutParams rootParams;
    private final LinearLayout.LayoutParams viewChildTextParams;
    private final LinearLayout.LayoutParams viewChildImageParams;

    // 方向
    private ToastOrientation currentToastOrientation = ToastOrientation.vertical;

    // 位置
    private int layoutGravity = Gravity.BOTTOM;
    private int layoutOfferX = 0;
    private int layoutOfferY = 200;


    private interface ToastParam {
        // 默认字体的大小 Toast字体大小
        float defaultTextSize = 14;
        // Toast字体颜色
        int defaultTextColor = 0XFFFFFFFF;
        // Toast背景颜色
        int defaultBgColor = R.drawable.bg_common_toast;

        // 最大行数
        int maxLine = 10;

        // 定义图片的大小
        int defaultImageSizeVertical = 40;
        int defaultImageSizeHorizontal = 30;

        int defaultTextMargin = 5;
        int defaultImageMargin = 5;
    }

    // 定义方向
    public enum ToastOrientation {
        vertical, horizontal
    }

    // 定义位置
    public enum ToastGravity {
        top, center, bottom
    }


    private CommonToast(Context context) {
        mContext = context;
        rootParams = new LinearLayout.LayoutParams(ViewGroup.LayoutParams.WRAP_CONTENT, ViewGroup.LayoutParams.WRAP_CONTENT);
        viewChildTextParams = new LinearLayout.LayoutParams(ViewGroup.LayoutParams.WRAP_CONTENT, ViewGroup.LayoutParams.WRAP_CONTENT);
        viewChildTextParams.setMargins(ToastParam.defaultTextMargin, ToastParam.defaultTextMargin, ToastParam.defaultTextMargin, ToastParam.defaultTextMargin);

        int size = 0;
        if (currentToastOrientation == ToastOrientation.vertical) {
            size = dp2px(ToastParam.defaultImageSizeVertical);
        } else {
            size = dp2px(ToastParam.defaultImageSizeHorizontal);
        }
        viewChildImageParams = new LinearLayout.LayoutParams(size, size);
        viewChildImageParams.setMargins(ToastParam.defaultImageMargin, ToastParam.defaultImageMargin, ToastParam.defaultImageMargin, ToastParam.defaultImageMargin);
    }

    public static CommonToast init() {
        if (mInstance == null) {
            synchronized (CommonToast.class) {
                if (mInstance == null) {
                    mInstance = new CommonToast(Utils.getApp().getApplicationContext());
                }
            }
        }
        return mInstance;
    }


    // 单例模式
    public static CommonToast init(Context context) {
        if (mInstance == null) {
            synchronized (CommonToast.class) {
                if (mInstance == null) {
                    mInstance = new CommonToast(context.getApplicationContext());
                }
            }
        }
        return mInstance;
    }

    private void getToast() {
        if (mToast == null) {
            mToast = new Toast(mContext);
            // 这里可以设置显示的位置 int gravity, int xOffset, int yOffset
            mToast.setGravity(layoutGravity, layoutOfferX, layoutOfferY);
            //设置显示时间
            mToast.setDuration(Toast.LENGTH_SHORT);
        }
    }

    //---------------------------------------------------------------------------------------------
    // 构建方向
    public CommonToast appendOrientation(ToastOrientation orientation) {
        this.currentToastOrientation = orientation;
        return this;
    }

    // 构建屏幕的位置
    public CommonToast appendGravity(ToastGravity gravity) {
        switch (gravity) {
            case top:
                layoutGravity = Gravity.TOP;
                break;
            case center:
                layoutGravity = Gravity.CENTER;
                break;
            case bottom:
                layoutGravity = Gravity.BOTTOM;
                break;
        }
        return this;
    }

    // 构建偏移量
    public CommonToast appendOfferX(int layoutOfferX) {
        this.layoutOfferX = layoutOfferX;
        return this;
    }

    // 构建偏移量
    public CommonToast appendOfferY(int layoutOfferY) {
        this.layoutOfferY = layoutOfferY;
        return this;
    }

    // 快捷设置（图文水平，居中展示）
    public CommonToast quickCenterHorizontal() {
        appendOrientation(ToastOrientation.horizontal);
        appendGravity(ToastGravity.center);
        this.layoutOfferX = 0;
        this.layoutOfferY = 0;
        return this;
    }

    // 快捷设置（图文垂直，居中展示）
    public CommonToast quickCenterVertical() {
        appendOrientation(ToastOrientation.vertical);
        appendGravity(ToastGravity.center);
        this.layoutOfferX = 0;
        this.layoutOfferY = 0;
        return this;
    }


    // 快捷设置（图文水平，底部展示）
    public CommonToast quickBottomHorizontal() {
        appendOrientation(ToastOrientation.horizontal);
        appendGravity(ToastGravity.bottom);
        this.layoutOfferX = 0;
        this.layoutOfferY = 200;
        return this;
    }

    // 快捷设置（图文垂直，底部展示）
    public CommonToast quickBottomVertical() {
        appendOrientation(ToastOrientation.vertical);
        appendGravity(ToastGravity.bottom);
        this.layoutOfferX = 0;
        this.layoutOfferY = 200;
        return this;
    }

    //---------------------------------------------------------------------------------------------


    public void show(String text) {
        makeText(mContext, text, 0);
    }

    public void show(int stringResourceId) {
        makeText(mContext, StringUtils.getString(stringResourceId), 0);
    }


    public void show(String text, int resourceImageId) {
        makeText(mContext, text, resourceImageId);
    }

    public void show(int stringResourceId, int resourceImageId) {
        makeText(mContext, StringUtils.getString(stringResourceId), resourceImageId);
    }


    private void makeText(Context context, String text, int resourceImageId) {
        init(context);
        getToast();
        if (mInstance.layout == null || mInstance.viewText == null) {
            LinearLayout container = new LinearLayout(mContext);
            container.setBackgroundResource(ToastParam.defaultBgColor);
            container.setLayoutParams(rootParams);
            container.setGravity(Gravity.CENTER);
            container.setOrientation(currentToastOrientation == ToastOrientation.horizontal ? LinearLayout.HORIZONTAL : LinearLayout.VERTICAL);

            mInstance.viewText = new TextView(mContext);
            mInstance.viewText.setLayoutParams(viewChildTextParams);
            mInstance.viewText.setPadding(dp2px(5), dp2px(2), dp2px(5), dp2px(2));
            mInstance.viewText.setGravity(Gravity.CENTER);
            mInstance.viewText.setTextColor(ToastParam.defaultTextColor);
            mInstance.viewText.setMaxLines(ToastParam.maxLine);
            mInstance.viewText.setTextSize(ToastParam.defaultTextSize);

            mInstance.viewImage = new ImageView(mContext);
            mInstance.viewImage.setLayoutParams(viewChildImageParams);
            mInstance.viewImage.setPadding(dp2px(5), dp2px(2), dp2px(5), dp2px(2));

            container.addView(mInstance.viewImage);
            container.addView(mInstance.viewText);
            mInstance.layout = container;
            mToast.setView(mInstance.layout);
        }

        mInstance.viewText.setText(text);
        mInstance.viewImage.setVisibility(resourceImageId == 0 ? View.GONE : View.VISIBLE);
        mInstance.viewImage.setImageResource(resourceImageId);
        mToast.show();
    }

    // dp转px
    private static int dp2px(float value) {
        float scale = mContext.getResources().getDisplayMetrics().density;
        return (int) (value * scale + 0.5f);
    }

    // 获得屏幕宽度
    private static int getScreenWidth(Context context) {
        WindowManager wm = (WindowManager) context.getSystemService(Context.WINDOW_SERVICE);
        DisplayMetrics outMetrics = new DisplayMetrics();
        wm.getDefaultDisplay().getMetrics(outMetrics);
        float density = outMetrics.density;
        return (int) (outMetrics.widthPixels * density);
    }
}
```
