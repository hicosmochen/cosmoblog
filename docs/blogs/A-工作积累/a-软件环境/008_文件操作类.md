# 文件操作类















```java
package com.study.java.util;

import java.io.*;
import java.nio.file.Files;
import java.nio.file.Paths;
import java.util.ArrayList;

public class CommonFileUtil {

    /***
     * 读取文件的内容到集合当中
     * @param filePath 文件的路径
     * @return 集合
     */
    public static ArrayList<String> readFileToArrayList(String filePath) {

        ArrayList<String> arrayList = new ArrayList<>();

        BufferedReader br = null;

        try {
            br = new BufferedReader(new InputStreamReader(Files.newInputStream(Paths.get(filePath))));
            String line = null;
            while ((line = br.readLine()) != null) {
                arrayList.add(line);
            }
        } catch (IOException e) {
            throw new RuntimeException(e);
        } finally {
            if (br != null) {
                try {
                    br.close();
                } catch (IOException e) {
                    throw new RuntimeException(e);
                }
            }
        }
        return arrayList;
    }


    /***
     * 将集合数据, 写入到文件当中。
     * @param arrayList 集合数据
     * @param filePath 写入文件的路径
     */
    public static void writeArrayListToFile(ArrayList<String> arrayList, String filePath) {
        if (arrayList == null) {
            return;
        }
        BufferedWriter bw = null;

        try {
            bw = new BufferedWriter(new OutputStreamWriter(Files.newOutputStream(Paths.get(filePath))));
            for (String element : arrayList) {
                bw.write(element);
                bw.newLine();
                bw.flush();
            }
        } catch (IOException e) {
            throw new RuntimeException(e);
        } finally {
            if (bw != null) {
                try {
                    bw.close();
                } catch (IOException e) {
                    throw new RuntimeException(e);
                }
            }
        }
    }

    /***
     * 序号填充
     * @param digit 位数
     * @param num 编号
     * @return 返回补零之后的结果
     */
    public static String getZeroNumber(int digit, int num) {
        String result = null;
        // 最大不超过 10 位数字
        if (digit <= 0 || digit > 10) {
            return result;
        }
        // 直接在前面补充0
        String oldResult = "0000000000" + num;
        // 直接对字符串进行分段截取。
        String oldResultAfter = oldResult.substring(oldResult.length() - digit);
        String oldResultBefore = oldResult.replace(oldResultAfter, "");

        // 100860 --> oldResultBefore = 00000000001
        // 100860 --> oldResultAfter = 00860
        // 查找 oldResultBefore 当中是否包含有不为0的数字
        int zeroNoIndex = -1;
        for (int i = 0; i < oldResultBefore.length(); i++) {
            if (oldResultBefore.charAt(i) != '0') {
                zeroNoIndex = i;
                break;
            }
        }
        // 找到了不为零的数据, 需要截取最新的数据
        if (zeroNoIndex >= 0) {
            String newResultBefore = oldResultBefore.substring(zeroNoIndex);
            result = newResultBefore + oldResultAfter;
        } else {
            result = oldResultAfter;
        }

        return result;
    }

    /**
     * 打印输出集合的信息
     *
     * @param arrayList       集合实体类对象
     * @param isPrintListSize 是否输出集合的长度
     */
    public static <T> void printArrayList(ArrayList<T> arrayList, boolean isPrintListSize) {
        if (arrayList == null) {
            return;
        }
        for (T element : arrayList) {
            System.out.println(element);
        }
        if (isPrintListSize) {
            System.out.println("-----> list.size: " + arrayList.size());
        }
    }
}
```
