## 第一章 前言

提出问题

```text
三个问题：
	1、为什么需要使用日志？
	2、为什么需要工具类？
	3、为什么需要自制工具类？
```

### 问题1：

```text
问题：
	为什么需要使用日志?
	
回答：
	日志是开发测试过程中, 针对于业务逻辑使用, 新增的一种复现 bug 追踪 bug 的一种手段。
	通过日志的复现, 去检查当前的业务逻辑是否存在漏洞或者错误情况。
	
简述:
	日志是追踪复现问题的一种手段
```

### 问题2：

```text
问题:
	为什么需要使用工具类?
	
回答:
	在我们采用日志的过程中, 会大量重复的使用日志, 对于重复的多次工作, 建议采用工具类, 快速完成日志的获取
	
简述:
	快速使用日志, 避免重复的操作
```

### 问题3：

```text
问题：
	为什么需要自制工具类?
	
回答:
	市面上存在多种日志操作的工具类, 但是每种工具类, 可能存在隐形的 bug 或其他问题。这种情况下, 需要自己去定义工具类

简述:
	自己定义减少bug 可以自由DIY 实现日志处理
```

```java
    private static void printLog(LogConfig localConfig, int priority, String tag, String message, Throwable throwable) {
        String safeMessage = (message != null) ? message : "null";
        List<ILogPrinter> printers = localConfig.printers;
        if (printers == null || printers.isEmpty()) return;

        String head = localConfig.isOpenHead ? formatLogHead(localConfig) : "";

        if (safeMessage.length() <= MAX_LOG_LENGTH) {
            for (ILogPrinter printer : printers) {
                try {
                    if (printer != null) {
                        printer.print(priority, tag, head, safeMessage, 0, safeMessage.length(), throwable);
                    }
                } catch (Throwable t) {
                    Log.e("CommonLogUtil", "Printer " + printer.getClass().getName() + " crash safety caught.", t);
                }
            }
        } else {
            int i = 0;
            int length = safeMessage.length();
            while (i < length) {
                int end = Math.min(i + MAX_LOG_LENGTH, length);

                while (end > i && end < length && Character.isHighSurrogate(safeMessage.charAt(end - 1))) {
                    end--;
                }
                if (end <= i) {
                    end = Math.min(i + MAX_LOG_LENGTH, length);
                }

                boolean isFirst = (i == 0);
                String currentHead = isFirst ? head : (localConfig.repeatHeadOnTruncate ? head : "[Cont.] ");

                for (ILogPrinter printer : printers) {
                    try {
                        if (printer != null) {
                            printer.print(priority, tag, currentHead, safeMessage, i, end, isFirst ? throwable : null);
                        }
                    } catch (Throwable t) {
                        Log.e("CommonLogUtil", "Printer " + printer.getClass().getName() + " crash safety caught on truncate.", t);
                    }
                }
                i = end;
            }
        }
    }

    private static String formatLogHead(LogConfig localConfig) {
        StringBuilder sb = new StringBuilder();
        if (localConfig.showBuildTime) sb.append("[").append(localConfig.buildTime).append("]");
        if (localConfig.showBuildType) sb.append("[").append(localConfig.buildType).append("]");
        if (localConfig.showBuildCode) sb.append("[").append(localConfig.buildCode).append("]");
        if (localConfig.showGlobalTag) sb.append("[").append(localConfig.globalTag).append("]");

        boolean isRelease = "RELEASE".equalsIgnoreCase(localConfig.buildType);
        boolean skipStackTrace = isRelease && localConfig.forceDisableStackTraceInRelease;
        boolean needStackTrace = !skipStackTrace && (localConfig.showThreadName || localConfig.showClassName || localConfig.showMethodName);

        if (localConfig.showThreadName) {
            sb.append("[").append(Thread.currentThread().getName()).append("]");
        }

        if (needStackTrace) {
            StackTraceElement[] stackTrace = new Throwable().getStackTrace();
            StackTraceElement targetElement = null;
            boolean foundLogUtil = false;

            int maxDepth = Math.min(stackTrace.length, 12);

            for (int i = 0; i < maxDepth; i++) {
                StackTraceElement element = stackTrace[i];
                String className = element.getClassName();
                if (className.startsWith(PACKAGE_PREFIX)
                        || className.startsWith("java.util.function.")
                        || className.startsWith("java.lang.reflect.")) {
                    foundLogUtil = true;
                    continue;
                }
                if (foundLogUtil) {
                    targetElement = element;
                    break;
                }
            }

            if (targetElement == null && stackTrace.length > 2) {
                targetElement = stackTrace[2];
            }

            if (targetElement != null) {
                if (localConfig.showClassName) {
                    String className = targetElement.getClassName();
                    int lastDot = className.lastIndexOf('.');
                    if (lastDot != -1) className = className.substring(lastDot + 1);

                    if (className.contains("$")) {
                        className = className.substring(0, className.indexOf("$"));
                    }

                    String fileName = targetElement.getFileName() != null ? targetElement.getFileName() : className + ".java";
                    sb.append("(").append(fileName).append(":").append(targetElement.getLineNumber()).append(")");
                }

                if (localConfig.showMethodName) {
                    String methodName = targetElement.getMethodName();
                    if (methodName.contains("$")) {
                        if (methodName.contains("$lambda$")) {
                            methodName = methodName.substring(0, methodName.indexOf("$lambda$"));
                        } else if (methodName.startsWith("lambda$")) {
                            String[] parts = methodName.split("\\$");
                            if (parts.length > 1) methodName = parts[1];
                        } else {
                            methodName = methodName.substring(0, methodName.indexOf("$"));
                        }
                    }
                    sb.append("[").append(methodName).append("]");
                }
            }
        }

        if (sb.length() > 0) sb.append(": ");
        return sb.toString();
    }
}
```

```kotlin
import android.util.Log
import java.util.concurrent.CopyOnWriteArrayList

interface ILogPrinter {
    fun print(priority: Int, tag: String?, fullMessage: String?, throwable: Throwable?)
}

class DefaultLogcatPrinter : ILogPrinter {
    override fun print(priority: Int, tag: String?, fullMessage: String?, throwable: Throwable?) {
        if (!fullMessage.isNullOrEmpty()) {
            Log.println(priority, tag, fullMessage)
        }
        if (throwable != null) {
            Log.println(priority, tag, Log.getStackTraceString(throwable))
        }
    }

    private fun printLog(
        localConfig: LogConfig, priority: Int, paramTag: String?, message: String?, throwable: Throwable?
    ) {
        val safeMessage = message ?: "null"
        val printers = localConfig.printers
        if (printers.isEmpty()) return

        // 这里做了 buildType 是 release 版本的校验, 如果是 release版本, 将会取消 类名称和方法名称的输出
        val isRelease = "RELEASE".equals(localConfig.buildType, ignoreCase = true)
        val skipStackTrace = isRelease && localConfig.forceDisableStackTraceInRelease
        val needStackTrace = !skipStackTrace && (localConfig.showThreadName || localConfig.showClassName ||
                localConfig.showMethodName || (paramTag == null && localConfig.autoTagFromStack))

        val targetElement = if (needStackTrace) findTargetStackTraceElement() else null
        val finalTag = paramTag ?: if (localConfig.autoTagFromStack && targetElement != null) {
            getShortClassName(targetElement.className)
        } else {
            localConfig.globalTag
        }

        val head = if (localConfig.isOpenHead) formatLogHead(localConfig, skipStackTrace, targetElement) else ""

        if (safeMessage.length <= MAX_LOG_LENGTH) {
            val fullMessage = if (head.isEmpty()) safeMessage else head + safeMessage
            printers.forEach { it.print(priority, finalTag, fullMessage, throwable) }
        } else {
            var i = 0
            val length = safeMessage.length
            var isFirstSegment = true
            var segmentCount = 0

            while (i < length) {
                if (segmentCount >= MAX_TRUNCATE_SEGMENTS) {
                    val fallbackMsg = if (localConfig.repeatHeadOnTruncate) head else ""
                    printers.forEach { it.print(priority, finalTag, "$fallbackMsg[Log Truncated: >${MAX_TRUNCATE_SEGMENTS} blocks]", null) }
                    break
                }

                var end = Math.min(i + MAX_LOG_LENGTH, length)
                while (end > i && end < length && Character.isHighSurrogate(safeMessage[end - 1])) end--
                if (end <= i) end = Math.min(i + MAX_LOG_LENGTH, length)

                val subMsg = safeMessage.substring(i, end)
                val currentHead = if (isFirstSegment) head else (if (localConfig.repeatHeadOnTruncate) head else "[Cont.] ")

                printers.forEach { it.print(priority, finalTag, currentHead + subMsg, if (isFirstSegment) throwable else null) }

                isFirstSegment = false
                segmentCount++
                i = end
            }
        }
    }

    private fun formatLogHead(localConfig: LogConfig, skipStackTrace: Boolean, element: StackTraceElement?): String {

        if (localConfig.showBuildType) sb.append("[").append(localConfig.buildType).append("]")
        if (localConfig.showBuildCode) sb.append("[").append(localConfig.buildCode).append("]")
        if (localConfig.showGlobalTag) sb.append("[").append(localConfig.globalTag).append("]")
        if (localConfig.showThreadName) sb.append("[").append(Thread.currentThread().name).append("]")

        if (!skipStackTrace && element != null) {
            if (localConfig.showClassName) {
                val className = getShortClassName(element.className)
                val fileName = element.fileName ?: "$className.class"
                val line = if (element.lineNumber >= 0) element.lineNumber.toString() else "Unknown"
                sb.append("(").append(fileName).append(":").append(line).append(")")
            }
            if (localConfig.showMethodName) sb.append("[").append(filterLambdaMethodName(element.methodName)).append("]")
        }
        if (sb.isNotEmpty()) sb.append(": ")
        return sb.toString()
    }

    private fun findTargetStackTraceElement(): StackTraceElement? {
        val stackTrace = Throwable().stackTrace
        val maxDepth = Math.min(stackTrace.size, 25)
        for (i in 0 until maxDepth) {
            val element = stackTrace[i]
            val className = element.className
            if (className.startsWith(CURRENT_PACKAGE_PREFIX) || className == CommonLogUtil::class.java.name) continue
            if (!className.startsWith("java.lang.") && !className.startsWith("dalvik.system.")) return element
        }
        return if (stackTrace.size > 2) stackTrace[2] else null
    }

    private fun getShortClassName(className: String): String {
        val lastDot = className.lastIndexOf('.')
        return if (lastDot != -1) className.substring(lastDot + 1) else className
    }

    private fun filterLambdaMethodName(method: String): String {
        var m = method
        if (m.contains("$")) {
            m = if (m.contains("\$lambda$")) m.substring(0, m.indexOf("\$lambda$"))
            else if (m.startsWith("lambda$")) {
                val f = m.indexOf('$'); val s = m.indexOf('$', f + 1)
                if (s != -1) m.substring(f + 1, s) else m.substring(f + 1)
            } else m.substring(0, m.indexOf("$"))
        }
        return m
    }
}
```

```text
日志的一些问题：性能的影响
虽然日志使用过程中, 能够快速帮助我们定位到问题所在地, 但是也会出现性能降低的问题,会进行大量的 IO 读写操作, 针对于 CPU 性能存在一定量的影响。
那么在使用的过程中, 需要有几点小的建议：
		1、日志需要精准, 能够通过当前的日志解决一类问题
		2、合理控制日志级别, 输出各类日志时, 考虑清楚日志是否需要
		3、正式环境中, 建议屏蔽掉日志信息, 可能存在日志包含有敏感信息的情况
```
