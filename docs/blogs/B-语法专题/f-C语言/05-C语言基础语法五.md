### 第01节 typedef 关键字

描述

```
typedef 关键字的作用:
	1、为一种数据类型(基本类型或自定义数据类型)定义一个新名字
	2、它并不能 创建新类型
	
typedef 和 define 有什么区别?
	1、作用对象不同：typedef仅限于数据类型，而不是能是表达式或具体的值
	2、作用时机不同: define发生在预处理，typedef发生在编译阶段
```

**案例代码**

```c
/**
    第一步 需要先在控制台中执行下面的代码, 进行编译操作
    gcc studyCDemo.c -o studyCDemo
    第二步 需要在控制台中执行下面的代码, 进行运行的操作
    studyCDemo
 */
#include <stdio.h> 

typedef int INT;
typedef char BYTE;
typedef BYTE T_BYTE;
typedef unsigned char UBYTE;
typedef struct type{
	UBYTE a;
	INT b;
	T_BYTE c;
} TYPE, *PTYPE;


int main(){
	
	TYPE tpye;
	tpye.a = 254;
	tpye.b = 10;
	tpye.c = 'c';
 
	PTYPE p = &tpye;
	
	printf("%u \n", p->a);
	printf("%d \n", p->b);
	printf("%c \n", p->c);
 
    return 0;
}
```

运行结果

> 254
>
> 10
>
> c







### 第02节 结构体定义

为什么会存在结构体类型？

```
数组的缺陷?
	定义：数组是一组具有相同数据类型的有序集合，可以用来处理大量相同类型的数据运算。
	缺陷：只能用于处理 "相同数据类型"
	
1、如果是不同的数据类型, 那么采用数组来进行处理，则无法实现了。
2、现实生活中，往往会存在不同数据类型的有机整体。 
3、例如: 学生。 一个学生有以下的类型: 学号、姓名、性别、年龄、地址 等属性
4、在上述内容中（学号、姓名、性别、年龄、地址） 如果采用统一的数据类型，则显得不够准确，也不便于管理。

因此, C语言中给出了另外一种数据类型 "结构体"
```

结构体如何定义？

```
定义结构体的方式有以下几种：
	1、先声明结构体类型，再定义变量名称
	2、在声明类型的同时 定义变量
	3、直接定义结构体类型的变量（无类型名）
```

举例说明

```c
// 方案一：定义结构体类型，再定义结构体名称。
// 结构体类型 stu   结构体变量名 Mike
struct stu {
    成员列表
};
struct stu MiKe;


// 方案二：在声明类型的同时 定义变量。 
// 结构体类型 stu  结构体变量名 Mike  Bob
struct stu {
    成员列表
} Mike, Bob;


// 方案三：直接定义结构体类型的变量（无类型名）  
// 结构体类型 无  结构体变量名 Mike  Bob
struct {
    成员列表
} Mike, Bob;
```

区分概念

```
结构体类型 和 结构体变量的关系：
	1、结构体类型：指定了结构体类型，它相当于一个模型，但是其中并没有具体数据，系统也不会分配实际的内存单元。  例如上面的 stu 
	2、结构体变量：系统根据结构体类型（内部成员的状况） 为之分配的空间。 例如上面的 Mike  Bob
```



**案例代码** 结构体类型的三种定义方式

```c
// 第一种定义结构体的方式
struct stu {
    char  name[50];
    int  age;
};
struct stu  s1 = { "mike", 18};


// 第二种定义结构体的方式
struct stu2 {
	char name[50];
	int age;
} s2 = {"lily", 22};


// 第三种定义结构体的方式
struct{
	char name[50];
	int age;
} s3 = {"yuri", 25};
```







### 第03节 结构体使用

**案例代码** 结构体的使用

```c
/**
    第一步 需要先在控制台中执行下面的代码, 进行编译操作
    gcc studyCDemo.c -o studyCDemo
    第二步 需要在控制台中执行下面的代码, 进行运行的操作
    studyCDemo
 */
#include <stdio.h> 

// 结构体的定义
struct stu {
    char  name[50];
    int  age;
};


int main(){
	struct stu s1;
	// 如果是普通变量, 通过点运算符操作结构体成员
	strcpy(s1.name, "abc");
	s1.age = 18;
	
	printf("s1.name = %s, s1.age = %d \n", s1.name, s1.age);
	
	// 如果是指针变量, 通过-> 操作结构体成员
	strcpy((&s1) -> name, "test");
	printf("(&s1)->name = %s, (&s1)->age = %d \n", (&s1)->name, (&s1)->age);
	
    return 0;
}
```

运行结果

> s1.name = abc, s1.age = 18
>
> (&s1)->name = test, (&s1)->age = 18

};

	// 定义一个包含有5个元素的结构体数组 并且将其初始化
	struct stu student[5] = {
		{101, "Li ping", 'M', 45},
		{102, "Zhang ping", 'M', 62.5},
		{103, "He fang", 'F', 92.5},

	avg = sum / length;          // 计算平均分
	printf("average = %f  count=%d \n", avg, count);

	for(i= 0; i<length; i++){
		printf("name = %s  score=%f \n", student[i].name, student[i].score);
	}

### 第06节 结构体链表

什么是链表？

```
1、链表是一种常用的数据结构，它通过指针将一些列数据结点，连接成一个数据链。相对于数组，链表具有更好的动态性（非顺序存储）链式存储。
2、数据域用来存储数据，指针域用于建立与下一个结点的联系。
3、建立链表时无需预先知道数据总量的，可以随机的分配空间，可以高效的在链表中的任意位置实时插入或删除数据。
4、链表的开销，主要是访问顺序性和组织链的空间损失。
```

链表和数组的区别？

| 对比项 | 数组                                                         | 链表                                                         |
| ------ | ------------------------------------------------------------ | ------------------------------------------------------------ |
| 概念   | 一次性分配一块连续的存储区域。                               | 无需一次性分配一块连续的存储区域，只需分配n块节点存储区域，通过指针建立关系。 |
| 优点   | 随机访问元素效率高                                           | 1、不需要一块连续的存储区域<br />2、删除和插入某个元素效率高 |
| 缺点   | 1、需要分配一块连续的存储区域（很大区域，有可能分配失败）<br />2、删除和插入某个元素效率低 | 随机访问元素效率低                                           |



**案例代码** C语言实现链表结构

```c
/**
    第一步 需要先在控制台中执行下面的代码, 进行编译操作
    gcc studyCDemo.c -o studyCDemo
    第二步 需要在控制台中执行下面的代码, 进行运行的操作
    studyCDemo
 */
#include <stdio.h>
#include <stdlib.h>
    
// 定义链表节点结构体    
typedef struct Node {
    int data;
    struct Node* next;    
} Node;
    
// 函数声明    
Node* createNode(int data);    
void insertAtHead(Node** head, int data);    
void insertAtTail(Node** head, int data);    
void deleteNode(Node** head, int data);    
void printList(Node* head);    
void freeList(Node** head);
    
int main() {
    Node* head = NULL;
    
    // 插入节点
    insertAtTail(&head, 10);
    insertAtTail(&head, 20);
    insertAtHead(&head, 5);
    insertAtTail(&head, 30);
    
    printf("初始链表: ");
    printList(head);
    
    // 删除节点
    deleteNode(&head, 20);
    printf("删除20后的链表: ");
    printList(head);
    
    // 再次插入
    insertAtHead(&head, 1);
    insertAtTail(&head, 40);
    printf("插入1和40后的链表: ");
    printList(head);
    
    // 释放链表内存
    freeList(&head);
    
    return 0;    
}
    
// 创建新节点    
Node* createNode(int data) {
    Node* newNode = (Node*)malloc(sizeof(Node));
    if (newNode == NULL) {
        printf("内存分配失败\n");
        exit(1);
    }
    newNode->data = data;
    newNode->next = NULL;
    return newNode;    
}
    
// 在链表头部插入节点    
void insertAtHead(Node** head, int data) {
    Node* newNode = createNode(data);
    newNode->next = *head;
    *head = newNode;    
}
    
// 在链表尾部插入节点    
void insertAtTail(Node** head, int data) {
    Node* newNode = createNode(data);
    
    if (*head == NULL) {
        *head = newNode;
        return;
    }
    
    Node* current = *head;
    while (current->next != NULL) {
        current = current->next;
    }
    current->next = newNode;    
}
    
// 删除指定数据的节点    
void deleteNode(Node** head, int data) {
    Node* temp = *head;
    Node* prev = NULL;
    
    if (temp != NULL && temp->data == data) {
        *head = temp->next;
        free(temp);
        return;
    }
    
    while (temp != NULL && temp->data != data) {
        prev = temp;
        temp = temp->next;
    }
    
    if (temp == NULL) {
        printf("未找到数据为%d的节点\n", data);
        return;
    }
    
    prev->next = temp->next;
    free(temp);    
}
    
// 打印链表    
void printList(Node* head) {
    Node* current = head;
    while (current != NULL) {
        printf("%d -> ", current->data);
        current = current->next;
    }
    printf("NULL\n");    
}
    
// 释放链表内存    
void freeList(Node** head) {
    Node* current = *head;
    Node* next;
    
    while (current != NULL) {
        next = current->next;
        free(current);
        current = next;
    }
    
    *head = NULL;
}
```

运行结果

> 初始链表: 5 -> 10 -> 20 -> 30 -> NULL
>
> 删除20后的链表: 5 -> 10 -> 30 -> NULL
>
> 插入1和40后的链表: 1 -> 5 -> 10 -> 30 -> 40 -> NULL























## 第二章  共同体

### 第01节 基础说明

```
1、联合union是一个能在同一个存储空间存储不同类型数据的类型；
2、联合体所占的内存长度等于其最长成员的长度倍数，也有叫做共用体；
3、同一内存段可以用来存放几种不同类型的成员，但每一瞬时只有一种起作用；
4、共用体变量中起作用的成员是最后一次存放的成员，在存入一个新的成员后原有的成员的值会被覆盖；
5、共用体变量的地址和它的各成员的地址都是同一地址。
```





### 第02节 案例代码

**案例代码** 共用体也叫联合体

```c
/**
    第一步 需要先在控制台中执行下面的代码, 进行编译操作
    gcc studyCDemo.c -o studyCDemo
    第二步 需要在控制台中执行下面的代码, 进行运行的操作
    studyCDemo
 */
#include <stdio.h>  

union Test{
	unsigned char  a;
	unsigned int   b;
	unsigned short c;
};

int main(){
	// 定义共用体变量
	union Test temp;
	// 1、所有成员的首地址是一样的
	printf("%p, %p, %p \n", &(temp.a), &(temp.b), &(temp.c));
	// 2、共用体大小为最大成员类型的大小
	printf("%lu \n", sizeof(union Test));
    
    // 3、一个成员赋值, 会影响另外的成员
	// 左边是高位,   右边是低位
	// 低位放低地址, 高位放高地址
	temp.b = 0x44332211;
	printf("%x \n", temp.a);
	printf("%x \n", temp.c);
	
	temp.a = 0x00;
	printf("short: %x \n", temp.c);
	printf("int: %x \n", temp.b);
    return 0;
}
```

运行结果

> 000000000061FE1C, 000000000061FE1C, 000000000061FE1C
>
> 4
>
> 11
>
> 2211
>
> short: 2200
>
> int: 44332200
